# Sample token definitions

# Whitespace includes a comment, which starts with '//' and
# extends to the end of the line; or, c-style comments /* ... */
#
# 0
WS:   ( [\f\r\s\t\n]+ ) \
    | ( // [^\n]* \n? ) \
    | ( /\*  ((\** [^*/] ) | [^*])*  \*+ / )

# c-style comment can be described as:
#
# '/*'
# followed by any number of either
#      any number of * followed by something other than * or /
#   or something other than *
# followed by one or more *
# followed by /
#

# An anonymous token, for convenience; a non-empty sequence of digits
#
_DIG: \d+

# Double has lower priority than int; we want ints to
# be interpreted as ints, not as doubles
#
# 1
#
DBL: \-?(({DIG}(.{DIG})?)|.{DIG})

# 2
INT: \-?{DIG}

# 3
LBL: '([^'\n]|\\')*'

# 4
ID:  [_a-zA-Z]\w*

# 5
ASSIGN: =

# 6
EQUIV: ==

# 7
IF: if

# 8
DO: do

# 9
BROP: \{

# 10
BRCL: \}
