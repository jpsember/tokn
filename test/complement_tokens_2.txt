# Let LF be a single linefeed, or a CR+LF combination
#
_LF: \0x0d? \0x0a

# Let SP be zero or more spaces or tabs (i.e., WS without linefeeds)
#
_SP: [\x09\0x20]*

# Let TX be any amount of text or whitespace, not including linefeeds
#
_TX: [^\0x0a\0x0d]*


_UN_ITEM: $SP [\+] $SP+ $TX

_NON_ITEM_TEXT: ^($UN_ITEM | $LF) $LF

IGNORED: ([\x09\0x20]+ $LF*) | $LF+

PARAGRAPH:  $SP $NON_ITEM_TEXT ($SP $NON_ITEM_TEXT)*

UNORDERED_ITEM: $UN_ITEM ($LF $SP $NON_ITEM_TEXT)*
